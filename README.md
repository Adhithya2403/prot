css
:root {
    --primary-color: rgb(163, 107, 215);    /* Purple accent */
    --secondary-color: rgb(35, 21, 85);     /* Dark blue */
    --dark-bg: rgb(0, 0, 33);              /* Dark background */
    --light-bg: rgb(22, 22, 60);           /* Light background */
